167  # Split the response text assuming the JSON objects are concatenated
168  json_strings = raw_response.split('}{')
169  #print('Number of JSON strings:', len(json_strings))
170  json_result_string = json_strings[0]
171  jsonObject = json.loads(json_result_string)
172  self.materialsmaterials.append(jsonObject)
173 
174  # Scan for next batch of materials
175  nextQuery = jsonObject['next']
176  if (nextQuery):
177  # Get limit and offset from this: 'https://api.matlib.gpuopen.com/api/materials/?limit=100&offset=100"'
178  # Split the string by '?'
179  queryParts = nextQuery.split('?')
180  # Split the string by '&'
181  queryParts = queryParts[1].split('&')
182  # Split the string by '='
183  limitParts = queryParts[0].split('=')
184  offsetParts = queryParts[1].split('=')
185  params['limit'] = int(limitParts[1])
186  params['offset'] = int(offsetParts[1])
187  print('Fetch set of materials: limit:', params['limit'], 'offset:', params['offset'])
188  else:
189  haveMoreMaterials = False
190  break
191 
192  dump = False
193  if dump:
194  count = self.materialsmaterials['count']
195  print('--------- Materials count:', count)
196  results = self.materialsmaterials['results']
